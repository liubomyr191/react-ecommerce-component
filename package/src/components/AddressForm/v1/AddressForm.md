### AddressForm


#### Basic Usage
```jsx
const countries = [
  { value: "US", label: "United State" },
  { value: "DE", label: "Germany" },
  { value: "NU", label: "Nigeria" }
];

const regions = [
  { value: "LA", label: "Llouisiana" },
  { value: "CA", label: "California" }
];

<AddressForm
  countries={countries}
  regions={regions}
 />
```
