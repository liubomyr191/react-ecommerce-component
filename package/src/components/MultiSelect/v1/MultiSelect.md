### Usage

```jsx
const options = [
  { value: 'mintchip', label: 'Mint Chip' },
  { value: 'chocolate', label: 'Chocolate' },
  { value: 'strawberry', label: 'Strawberry' },
  { value: 'vanilla', label: 'Vanilla' },
  { value: 'darkchocolate', label: 'Dark Chocolate' }
];

<MultiSelect options={options} />
```
